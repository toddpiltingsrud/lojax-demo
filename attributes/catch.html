jx-catch

Supply the name of a callback function to execute when an AJAX call fails.
Complex paths are supported (e.g. fns.errorCallback). 

<script>
    var fns = fns || {};
    fns.errorCallback = function ( err ) {
        $( '[jx-panel=right] pre' ).fadeIn();
    };
</script>

<pre class="well" style="display:none;margin-top:20px;">
    <button type="button" 
        class="btn btn-primary" 
        jx-method="ajax-get" 
        jx-action="attributes/doesnotexist.html" 
        jx-catch="fns.errorCallback">
        This button specifies "<code>fns.errorCallback</code>" as a failure callback.
    </button>    
</pre>
